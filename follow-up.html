  <script>

    let tasks = JSON.parse(localStorage.getItem('tasks') || '[]');


    function renderTasks() {

      const tbody = document.getElementById('taskList');

      tbody.innerHTML = '';

      tasks.forEach((task, i) => {

        tbody.innerHTML += `

          <tr>

            <td>${task.title}</td>

            <td>${task.building}</td>

            <td>${task.floor}</td>

            <td>${task.assignee}</td>

            <td>

              <span class="status ${task.status === 'منجزة' ? 'done' : 'todo'}">${task.status}</span>

            </td>

            <td>

              <button onclick="markDone(${i})" style="background:#10b981">${task.status === 'منجزة' ? '✔️' : 'إنهاء'}</button>

              <button onclick="delTask(${i})" style="background:#ef4444">حذف</button>

            </td>

          </tr>

        `;

      });

    }


    document.getElementById('taskForm').onsubmit = function(e) {

      e.preventDefault();

      const title = document.getElementById('title').value.trim();

      const building = document.getElementById('building').value;

      const floor = document.getElementById('floor').value;

      const assignee = document.getElementById('assignee').value;

      if (!title || !assignee || !building || !floor) return;

      tasks.push({ title, building, floor, assignee, status: 'قيد التنفيذ' });

      localStorage.setItem('tasks', JSON.stringify(tasks));

      renderTasks();

      document.getElementById('taskForm').reset();

      alert(`تم تعيين المهمة "${title}" (${building}، الدور ${floor}) إلى ${assignee}.\n`);

    };


    function markDone(i) {

      if (tasks[i].status !== 'منجزة') {

        tasks[i].status = 'منجزة';

        localStorage.setItem('tasks', JSON.stringify(tasks));

        renderTasks();

      }

    }


    function delTask(i) {

      if (confirm('هل أنت متأكد من حذف المهمة؟')) {

        tasks.splice(i, 1);

        localStorage.setItem('tasks', JSON.stringify(tasks));

        renderTasks();

      }

    }


    renderTasks();

  </script>
